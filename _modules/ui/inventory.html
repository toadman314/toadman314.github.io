    def toggle(self):
        """Toggle the visibility of the Inventory UI."""
        self.visible = not self.visible
        self.selected_index = None
        self.dragging = None
        self.context_menu = None
        self.split_prompt = None

    def handle_event(self, event, player, game_map):
        """Handle events for the Inventory UI."""
        if not self.visible:
            return

        # Handle split prompt input first
        if self.split_prompt:
            if event.type == pygame.KEYDOWN:
                if event.key == pygame.K_RETURN:
                    amount = int(self.split_prompt[2]) if self.split_prompt[2].isdigit() else 0
                    if 1 <= amount <= self.split_prompt[1]:
                        self._split_stack(self.split_prompt[0], amount)
                        self.split_prompt = None
                elif event.key == pygame.K_ESCAPE:
                    self.split_prompt = None
                elif event.unicode.isdigit():
                    # Prevent leading zeros
                    if not (event.unicode == "0" and self.split_prompt[2] == ""):
                        self.split_prompt[2] += event.unicode
                elif event.key == pygame.K_BACKSPACE:
                    self.split_prompt[2] = self.split_prompt[2][:-1]
            return  # Don't process other events while prompt is active

        mouse_pos = pygame.mouse.get_pos()
        inv_rect = pygame.Rect(40, 80, self.cols * self.cell_size, self.rows * self.cell_size)
        equip_rect = pygame.Rect(inv_rect.right + 40, 80, self.equip_slot_size * 3, self.equip_slot_size * ((len(self.equip_slots) + 2) // 3))

        if event.type == pygame.MOUSEBUTTONDOWN:
            if event.button == 1:  # Left click
                idx = self._slot_at_pos(mouse_pos, inv_rect)
                if idx is not None and idx < len(self.player.inventory):
                    self.dragging = (idx, None)
                    self.selected_index = idx
                    self.context_menu = None
                else:
                    eq = self._equip_at_pos(mouse_pos, equip_rect)
                    if eq:
                        self.selected_index = None
                        self.hovered_equip = eq
                        self.context_menu = None
            elif event.button == 3:  # Right click
                idx = self._slot_at_pos(mouse_pos, inv_rect)
                if idx is not None and idx < len(self.player.inventory):
                    slot = self.player.inventory[idx]
                    actions = slot.item.get_context_actions() if slot.item else []
                    if slot.item and slot.item.stackable and slot.count > 1:
                        actions.append("Split Stack")
                    self.context_menu = (idx, actions)
                else:
                    eq = self._equip_at_pos(mouse_pos, equip_rect)
                    if eq and self.player.equipment[eq]:
                        self.context_menu = (eq, ["Unequip"])
        elif event.type == pygame.MOUSEBUTTONUP:
            if self.dragging:
                idx_from, split = self.dragging
                idx_to = self._slot_at_pos(mouse_pos, inv_rect)
                if idx_to is not None and idx_to != idx_from:
                    self._move_or_split(idx_from, idx_to, split)
                else:
                    eq = self._equip_at_pos(mouse_pos, equip_rect)
                    if eq:
                        self.player.equip_item(idx_from)
                self.dragging = None
        elif event.type == pygame.MOUSEMOTION:
            self.hovered_index = self._slot_at_pos(mouse_pos, inv_rect)
            self.hovered_equip = self._equip_at_pos(mouse_pos, equip_rect)
        elif event.type == pygame.KEYDOWN:
            if event.key == pygame.K_ESCAPE:
                self.toggle()
            elif event.key == pygame.K_DELETE and self.selected_index is not None:
                self.player.drop_item(self.selected_index, game_map)
                self.selected_index = None

    def handle_context_action(self, action, idx, player, game_map):
        """Handle context menu actions."""
        if action == "Drop":
            player.drop_item(idx, game_map)
        elif action == "Equip":
            player.equip_item(idx)
        elif action == "Split Stack":
            slot = player.inventory[idx]
            if slot.count > 1:
                self.split_prompt = [idx, slot.count - 1, ""]
        elif action == "Consume":
            player.use_item(idx)
        elif action == "Unequip":
            slot = idx
            item = player.equipment[slot]
            if item:
                player.add_item(item, 1)
                player.equipment[slot] = None

    def draw(self, screen):
        """
        Draw the Inventory UI panel with the graphical background and overlays.

        Parameters
        ----------
        screen : pygame.Surface
            The display surface to draw the inventory panel on.
        """
        if not self.visible:
            return
        # Draw the inventory_panel.png background centered and at full size
        width, height = self.bg_image.get_width(), self.bg_image.get_height()
        sw, sh = screen.get_size()
        x = (sw - width) // 2
        y = (sh - height) // 2
        screen.blit(self.bg_image, (x, y))

        # Inventory grid and equipment overlays (drawn on top)
        inv_rect = pygame.Rect(40, 80, self.cols * self.cell_size, self.rows * self.cell_size)
        for i in range(self.cols * self.rows):
            x = inv_rect.x + (i % self.cols) * self.cell_size
            y = inv_rect.y + (i // self.cols) * self.cell_size
            rect = pygame.Rect(x, y, self.cell_size, self.cell_size)
            pygame.draw.rect(screen, (80, 80, 120), rect, 2)
            if i < len(self.player.inventory):
                slot = self.player.inventory[i]
                if slot.item:
                    pygame.draw.rect(screen, (180, 180, 80), rect.inflate(-8, -8))
                    if slot.item.stackable and slot.count > 1:
                        count_surf = self.font.render(str(slot.count), True, (255, 255, 255))
                        screen.blit(count_surf, (rect.right - count_surf.get_width() - 4, rect.bottom - count_surf.get_height() - 4))
                    if self.selected_index == i:
                        pygame.draw.rect(screen, (255, 255, 100), rect, 3)
            if self.hovered_index == i:
                pygame.draw.rect(screen, (120, 200, 255), rect, 2)

        equip_rect = pygame.Rect(inv_rect.right + 40, 80, self.equip_slot_size * 3, self.equip_slot_size * ((len(self.equip_slots) + 2) // 3))
        pygame.draw.rect(screen, (60, 60, 90), equip_rect)
        for idx, slot in enumerate(self.equip_slots):
            col = idx % 3
            row = idx // 3
            x = equip_rect.x + col * self.equip_slot_size
            y = equip_rect.y + row * self.equip_slot_size
            rect = pygame.Rect(x, y, self.equip_slot_size, self.equip_slot_size)
            pygame.draw.rect(screen, (120, 120, 160), rect, 2)
            item = self.player.equipment.get(slot)
            if item:
                pygame.draw.rect(screen, (200, 160, 80), rect.inflate(-8, -8))
                name_surf = self.font.render(item.name, True, (255, 255, 255))
                screen.blit(name_surf, (rect.x + 4, rect.y + 4))
            if self.hovered_equip == slot:
                pygame.draw.rect(screen, (120, 200, 255), rect, 2)

        if self.context_menu:
            idx, actions = self.context_menu
            mx, my = pygame.mouse.get_pos()
            menu_rect = pygame.Rect(mx, my, 120, 24 * len(actions))
            pygame.draw.rect(screen, (30, 30, 30), menu_rect)
            for i, action in enumerate(actions):
                act_surf = self.font.render(action, True, (255, 255, 255))
                screen.blit(act_surf, (menu_rect.x + 8, menu_rect.y + i * 24 + 4))

        if self.split_prompt:
            idx, max_amt, input_str = self.split_prompt
            prompt_rect = pygame.Rect(300, 300, 220, 80)
            pygame.draw.rect(screen, (30, 30, 30), prompt_rect)
            pygame.draw.rect(screen, (200, 200, 100), prompt_rect, 2)
            msg = f"Split how many? (1-{max_amt})"
            msg_surf = self.font.render(msg, True, (255, 255, 255))
            input_surf = self.font.render(input_str, True, (255, 255, 100))
            screen.blit(msg_surf, (prompt_rect.x + 10, prompt_rect.y + 10))
            screen.blit(input_surf, (prompt_rect.x + 10, prompt_rect.y + 40))

        if self.hovered_index is not None and self.hovered_index < len(self.player.inventory):
            slot = self.player.inventory[self.hovered_index]
            if slot.item:
                tooltip = f"{slot.item.name} x{slot.count}" if slot.item.stackable else slot.item.name
                tip_surf = self.font.render(tooltip, True, (255, 255, 200))
                screen.blit(tip_surf, (pygame.mouse.get_pos()[0] + 16, pygame.mouse.get_pos()[1] + 8))
        elif self.hovered_equip:
            item = self.player.equipment.get(self.hovered_equip)
            if item:
                tip_surf = self.font.render(item.name, True, (255, 255, 200))
                screen.blit(tip_surf, (pygame.mouse.get_pos()[0] + 16, pygame.mouse.get_pos()[1] + 8))

        # Draw message (for errors like "no empty slot")
        if self.message:
            msg_surf = self.font.render(self.message, True, (255, 80, 80))
            screen.blit(msg_surf, (40, 60))

    def _slot_at_pos(self, pos, inv_rect):
        mx, my = pos
        if not inv_rect.collidepoint(mx, my):
            return None
        col = (mx - inv_rect.x) // self.cell_size
        row = (my - inv_rect.y) // self.cell_size
        idx = row * self.cols + col
        if 0 <= idx < self.cols * self.rows:
            return idx
        return None

    def _equip_at_pos(self, pos, equip_rect):
        mx, my = pos
        if not equip_rect.collidepoint(mx, my):
            return None
        col = (mx - equip_rect.x) // self.equip_slot_size
        row = (my - equip_rect.y) // self.equip_slot_size
        idx = row * 3 + col
        if 0 <= idx < len(self.equip_slots):
            return self.equip_slots[idx]
        return None

    def _split_stack(self, idx, amount):
        slot = self.player.inventory[idx]
        if slot.item and slot.count > amount:
            # Find first empty slot
            empty_idx = None
            for i in range(len(self.player.inventory)):
                if self.player.inventory[i].item is None:
                    empty_idx = i
                    break
            if empty_idx is not None:
                slot.count -= amount
                self.player.inventory[empty_idx].item = slot.item
                self.player.inventory[empty_idx].count = amount
                self.message = f"Split {amount} {slot.item.name}(s) to slot {empty_idx+1}."
            else:
                self.message = "No empty slot available to split stack!"
        else:
            self.message = "Invalid split amount."
