    def draw(self, screen):
        """
        Draw the Character UI panel with the graphical background and player stats.
        """
        if not self.visible:
            return
        width, height = self.bg_image.get_width(), self.bg_image.get_height()
        sw, sh = screen.get_size()
        x = (sw - width) // 2
        y = (sh - height) // 2
        screen.blit(self.bg_image, (x, y))

        # --- Draw BASIC, INNATE, and LEARNED stats as string placeholders ---
        stat_font = pygame.font.SysFont(None, 28)
        group_offsets = {
            'BASIC': (x + 40, y + 60),
            'INNATE': (x + 260, y + 60),
            'LEARNED': (x + 480, y + 60),
        }
        # Draw stats with effects applied
        for group, offset in group_offsets.items():
            if group == 'BASIC':
                stats = self.player._basic_stats
            elif group == 'INNATE':
                stats = self.player._innate_stats
            else:
                stats = self.player._learned_stats
            y_offset = 0
            for stat, base in stats.items():
                mod = self.player.get_modified_stat(stat, base)
                stat_str = f"{stat.capitalize()}: {mod}"
                surf = stat_font.render(stat_str, True, (0,0,0))
                screen.blit(surf, (offset[0], offset[1] + y_offset))
                y_offset += 28
