    def draw(self, screen):
        """Draw the Dialogue UI panel."""
        if not self.visible:
            return
        width, height = 600, 200
        sw, sh = screen.get_size()
        x = (sw - width) // 2
        y = sh - height - 40
        panel_rect = pygame.Rect(x, y, width, height)
        pygame.draw.rect(screen, (40, 60, 40), panel_rect)
        pygame.draw.rect(screen, (200, 255, 200), panel_rect, 3)
        text = self.font.render("Dialogue (Placeholder)", True, (255, 255, 200))
        screen.blit(text, (x + 40, y + 40))
