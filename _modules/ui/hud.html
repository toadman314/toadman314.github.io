    def _update_action_button_rects(self, screen=None):
        """
        Calculate and update the rectangles for action buttons based on the current screen size.

        Parameters
        ----------
        screen : pygame.Surface, optional
            The display surface. If not provided, uses the last known size or UI manager's screen.
        """
        if screen is not None:
            sw, sh = screen.get_size()
        else:
            # Fallback: try to get from ui_manager or default
            sw, sh = 800, 600
            if hasattr(self.ui_manager, 'screen') and self.ui_manager.screen:
                sw, sh = self.ui_manager.screen.get_size()
        x = sw - HUD_WIDTH
        action_start_y = 340  # Just below permanent buttons, adjust as needed
        action_x0 = x + 8
        action_x1 = x + 8 + 36  # 32px + 4px gap
        self.action_button_rects.clear()
        for btn in ACTION_BUTTONS:
            col = btn.get("col", 0)
            row = btn.get("row", 0)
            btn_x = action_x0 if col == 0 else action_x1
            btn_y = action_start_y + row * 36
            rect = pygame.Rect(btn_x, btn_y, 32, 32)
            self.action_button_rects[btn["name"]] = rect

    def draw(self, screen):
        """
        Draw the HUD, including permanent and action buttons, and player info.

        Parameters
        ----------
        screen : pygame.Surface
            The display surface to draw the HUD on.
        """
        self._update_action_button_rects(screen)
        sw, sh = screen.get_size()
        x = sw - HUD_WIDTH
        y = 0
        # Draw the HUD background image, scaled to fit the right side
        bg = pygame.transform.scale(self.bg_image, (HUD_WIDTH, sh))
        screen.blit(bg, (x, y))

        # --- Phase Display ---
        phase = self.ui_manager.get_phase() if self.ui_manager else None
        phase_str = {
            "PLAYER_TURN": "Player Turn",
            "NPC_TURN": "NPC Turn",
            "ANIMAL_TURN": "Animal Turn",
            "END_OF_TURN": "End of Turn",
            "UPDATE_AFTER_PLAYER": "...",
            "UPDATE_AFTER_NPC": "...",
            "UPDATE_AFTER_ANIMAL": "...",
            "MENU": "Menu"
        }.get(phase, "")
        if phase_str:
            surf = self.font.render(phase_str, True, (255, 255, 0))
            screen.blit(surf, (x + 12, 8))

        # --- Info Section ---
        info_y = 20 + (32 if phase_str else 0)
        # Time of Day
        time_str = getattr(self.game_state, "time_of_day", "Unknown")
        surf = self.font.render(f"Time: {time_str}", True, (255, 255, 200))
        screen.blit(surf, (x + 20, info_y))
        info_y += 32

        # Action Points
        ap = getattr(self.player, "action_points", 0)
        max_ap = getattr(self.player, "max_action_points", 0)
        surf = self.font.render(f"AP: {ap} / {max_ap}", True, (200, 255, 200))
        screen.blit(surf, (x + 20, info_y))
        info_y += 32

        # Health Condition
        health = getattr(self.player, "health_condition", "Unknown")
        surf = self.font.render(f"Health: {health}", True, (255, 200, 200))
        screen.blit(surf, (x + 20, info_y))
        info_y += 32

        # Money
        money = getattr(self.player, "money", 0)
        surf = self.font.render(f"Gold: {money}", True, (255, 255, 100))
        screen.blit(surf, (x + 20, info_y))
        info_y += 32

        # Weight
        weight = getattr(self.player, "weight", 0)
        weight_limit = getattr(self.player, "weight_limit", 0)
        surf = self.font.render(f"Weight: {weight} / {weight_limit}", True, (200, 200, 255))
        screen.blit(surf, (x + 20, info_y))
        info_y += 32

        # --- Buttons Section ---
        self.button_rects.clear()
        is_player_turn = (phase == "PLAYER_TURN")
        # Draw permanent HUD buttons
        for btn in HUD_BUTTONS:
            rect = pygame.Rect(x + 6, btn["y"], 84, 32)
            self.button_rects[btn["name"]] = rect
            # Determine button state: 0 = unavailable, 1 = available, 2 = toggled
            if btn["ui_attr"] is not None:
                ui_panel = getattr(self.ui_manager, btn["ui_attr"], None)
                if ui_panel is None:
                    state = 0
                elif getattr(ui_panel, "visible", False):
                    state = 2
                else:
                    state = 1
            else:
                state = 1  # All remaining permanent buttons are always enabled
            src_rect = pygame.Rect(state * 84, 0, 84, 32)
            screen.blit(self.button_images[btn["name"]], rect.topleft, src_rect)
        # Draw action buttons
        self._update_action_button_rects(screen)
        for btn in ACTION_BUTTONS:
            rect = self.action_button_rects[btn["name"]]
            img = self.action_button_images[btn["name"]]
            screen.blit(img, rect.topleft)
        # --- Tooltip ---
        # Show tooltip for hovered button (permanent or action)
        if self.hovered_button:
            if self.hovered_button in self.button_rects:
                btn = next(b for b in HUD_BUTTONS if b["name"] == self.hovered_button)
                mx, my = pygame.mouse.get_pos()
                tip_surf = self.font.render(btn["tooltip"], True, (255, 255, 200))
                tip_rect = tip_surf.get_rect(topleft=(mx + 16, my + 8))
                pygame.draw.rect(screen, (30, 30, 30), tip_rect.inflate(12, 8))
                screen.blit(tip_surf, tip_rect)
            elif self.hovered_button in self.action_button_rects:
                btn = next(b for b in ACTION_BUTTONS if b["name"] == self.hovered_button)
                mx, my = pygame.mouse.get_pos()
                tip_surf = self.font.render(btn["tooltip"], True, (255, 255, 200))
                tip_rect = tip_surf.get_rect(topleft=(mx + 16, my + 8))
                pygame.draw.rect(screen, (30, 30, 30), tip_rect.inflate(12, 8))
                screen.blit(tip_surf, tip_rect)

    def handle_event(self, event, allow_action_buttons=True):
        """
        Handle a Pygame event and route it to the appropriate HUD button or action button.

        Parameters
        ----------
        event : pygame.event.Event
            The event to handle.
        allow_action_buttons : bool, optional
            Whether to allow action button events (default: True).
        """
        self._update_action_button_rects()
        if event.type == pygame.MOUSEMOTION:
            self.hovered_button = None
            mx, my = event.pos
            for name, rect in {**self.button_rects, **self.action_button_rects}.items():
                if rect.collidepoint(mx, my):
                    self.hovered_button = name
                    break
        elif event.type == pygame.MOUSEBUTTONDOWN and event.button == 1:
            mx, my = event.pos
            # Always allow permanent HUD buttons
            for name, rect in self.button_rects.items():
                if rect.collidepoint(mx, my):
                    self._on_button_click(name)
                    return
            # Only allow action buttons if allowed
            if allow_action_buttons:
                for name, rect in self.action_button_rects.items():
                    if rect.collidepoint(mx, my):
                        self._on_button_click(name)
                        return



    def _on_button_click(self, name):
        """
        Handle a button click by name, routing to the appropriate action or toggle method.

        Parameters
        ----------
        name : str
            The name of the button clicked.
        """
        # Permanent HUD buttons
        if name in [btn["name"] for btn in HUD_BUTTONS]:
            btn = next((b for b in HUD_BUTTONS if b["name"] == name), None)
            if btn:
                if btn.get("toggle_method") and hasattr(self.ui_manager, btn["toggle_method"]):
                    getattr(self.ui_manager, btn["toggle_method"])()
            return
        # Action buttons
        if name in [btn["name"] for btn in ACTION_BUTTONS]:
            self._handle_action_button(name)
            return
