    def choose_interaction(self, target, encounter_type):
        """
        Determines which interaction to use based on the encounter type.

        Parameters
        ----------
        target : object
            The target entity or item.
        encounter_type : str
            The type of encounter ("entity", "item", or "object").

        Returns
        -------
        object or None
            The chosen interaction object, or None if no valid interaction.
        """
        if encounter_type == "entity":
            # Placeholder: If NPC, offer trade/talk, else attack
            if hasattr(target, "__class__") and target.__class__.__name__ == "NPC":
                return DialogueInteraction()
            else:
                return KillEntityInteraction()
        elif encounter_type == "item":
            return PickUpItemInteraction(target)
        elif encounter_type == "object":
            return DisengageEntityInteraction()
        return None

    def add_item(self, item, count=1):
        """
        Adds an item (with optional count) to the inventory, handling stacking.

        Parameters
        ----------
        item : Item
            The item to add.
        count : int, optional
            The number of items to add (default is 1).
        """
        if not item:
            logger.warning("Tried to add None item to inventory.")
            return
        if getattr(item, "stackable", False):
            for slot in self.inventory:
                if slot.item.name == item.name and slot.count < item.max_stack:
                    addable = min(count, item.max_stack - slot.count)
                    if addable <= 0:
                        if hasattr(self, "game") and hasattr(self.game, "events"):
                            self.game.events.post_immediate(InventoryFullEvent(self, item))
                        continue
                    slot.count += addable
                    count -= addable
                    if hasattr(self, "game") and hasattr(self.game, "events"):
                        self.game.events.post_immediate(StackChangedEvent(self, item, slot.count))
                    if count == 0:
                        return
            while count > 0:
                addable = min(count, item.max_stack)
                self.inventory.append(InventorySlot(item, addable))
                if hasattr(self, "game") and hasattr(self.game, "events"):
                    self.game.events.post_immediate(StackChangedEvent(self, item, addable))
                count -= addable
        else:
            for _ in range(count):
                self.inventory.append(InventorySlot(item, 1))
                if hasattr(self, "game") and hasattr(self.game, "events"):
                    self.game.events.post_immediate(StackChangedEvent(self, item, 1))

    def remove_item(self, slot_index, count=1):
        """
        Removes a specified count of items from a given inventory slot.

        Parameters
        ----------
        slot_index : int
            The index of the inventory slot.
        count : int, optional
            The number of items to remove (default is 1).
        """
        if slot_index < 0 or slot_index >= len(self.inventory):
            logger.warning(f"Tried to remove item from invalid inventory index {slot_index}.")
            return
        slot = self.inventory[slot_index]
        if count > slot.count:
            logger.warning(f"Tried to remove {count} items from a stack of {slot.count}.")
            count = slot.count
        if slot.count > count:
            slot.count -= count
        else:
            self.inventory.pop(slot_index)

    def drop_item(self, slot, game_map):
        """
        Drops an item from the inventory onto the map at the player's position.

        idx = self.inventory.index(slot)
        self.remove_item(idx, 1)
        game_map.items[(self.x, self.y)] = slot.item

    def equip_item(self, slot_index):
        """
        Equips an item from the inventory to the appropriate equipment slot.
        Swaps out any currently equipped item.

        Parameters
        ----------
        slot_index : int
            The index of the inventory slot containing the item to equip.
        """
        if slot_index < 0 or slot_index >= len(self.inventory):
            logger.warning(f"Tried to equip from invalid inventory index {slot_index}.")
            return
        slot = self.inventory[slot_index]
        item = slot.item
        if not item.slot_type:
            logger.info(f"Cannot equip {item.name}: not an equippable item.")
            return
        if item.slot_type not in self.equipment:
            logger.info(f"Cannot equip {item.name}: invalid equipment slot '{item.slot_type}'.")
            return
        currently_equipped = self.equipment[item.slot_type]
        self.equipment[item.slot_type] = item
        self.remove_item(slot_index, 1)
        if hasattr(self, "game") and hasattr(self.game, "events"):
            self.game.events.post_immediate(ItemEquippedEvent(self, item, item.slot_type))
        if currently_equipped:
            if hasattr(self, "game") and hasattr(self.game, "events"):
                self.game.events.post_immediate(ItemUnequippedEvent(self, currently_equipped, item.slot_type))
            self.add_item(currently_equipped, 1)

    def confirm_move(self, game_map):
        """
        Move along the planned path, consuming action points (cost 1 for orthogonal, 1.5 for diagonal).

        Parameters
        ----------
        game_map : GameMap
            The game map instance, used to check passability of tiles.

        Returns
        -------
        int
            The number of steps successfully moved.
        """
        steps = 0
        while self.planned_path and self.action_points > 0:
            next_x, next_y = self.planned_path[0]
            step_cost = self.planned_path_costs[0] if self.planned_path_costs else 1
            if not game_map.is_passable(next_x, next_y):
                break
            if self.action_points < step_cost:
                break
            self.x, self.y = next_x, next_y
            self.planned_path.pop(0)
            self.planned_path_costs.pop(0)
            self.action_points -= step_cost
            steps += 1
            if hasattr(game_map, 'on_player_step'):
                game_map.on_player_step(self.x, self.y, self)
        return steps
