    def get_context_actions(self):
        """
        Returns a list of context menu actions available for this item.

        Returns
        -------
        list
            List of action names as strings.
        """
        actions = ["Drop"]
        if self.stackable:
            actions.append("Split Stack")
        if self.slot_type:
            actions.append("Equip")
        return actions
